# Audio DiT — Rectified Flow Training (unconditional)
#
# Install order:
#   1. Create venv
#       python -m venv .venv
#       source .venv/bin/activate         # Linux/macOS
#       .\.venv\Scripts\Activate.ps1      # Windows PowerShell
#
#   2. Install PyTorch with CUDA support
#      Pick the cu* tag matching your CUDA DRIVER:
#        - CUDA 12.x driver:  cu124  (e.g. the older IRCAM 12 GB GPUs)
#            pip install torch==2.6.0 torchvision==0.21.0 torchaudio==2.6.0 --index-url https://download.pytorch.org/whl/cu124
#        - CUDA 13.x driver:  cu130  (e.g. chichibu / RTX A5000, driver 13.x)
#            pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu130
#
#   3. Install everything else
#       pip install -r requirements.txt
#
#   4. Verify
#       python -c "import torch; print(torch.cuda.is_available())"
#
# Tested on Python 3.10, 3.11, 3.12 (Linux + Windows).
